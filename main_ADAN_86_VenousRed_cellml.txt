    def comp Pulmonary as
        var t: second {pub: in};

        // Constitutive Parameters

        var C_pas: m6_per_J {priv: in};
        var C_pat: m6_per_J {priv: in};
        var C_par: m6_per_J {priv: in};
        var C_pcp: m6_per_J {priv: in};
        var C_pvn: m6_per_J {priv: in};
        var C_pvc: m6_per_J {priv: in};
        var R_pas: J_s_per_m6 {priv: in};
        var R_pat: J_s_per_m6 {priv: in};
        var R_par: J_s_per_m6 {priv: in};
        var R_pcp: J_s_per_m6 {priv: in};
        var R_psh: J_s_per_m6 {priv: in};
        var R_pvn: J_s_per_m6 {priv: in};
        var I_pas: J_s2_per_m6 {priv: in};
        var I_pat: J_s2_per_m6 {priv: in};
        var I_par: J_s2_per_m6 {priv: in};
        var I_pcp: J_s2_per_m6 {priv: in};
        var I_pvn: J_s2_per_m6 {priv: in};
        var I_psh: J_s2_per_m6 {priv: in};

        // State Variables

        var u_pas: J_per_m3 {init: 4000.0, pub: out};
        var u_pat: J_per_m3 {init: 0.0, pub: out};
        var u_par: J_per_m3 {init: 0.0, pub: out};
        var u_pcp: J_per_m3 {init: 0.0, pub: out};
        var u_pvn: J_per_m3 {init: 0.0, pub: out};
        var u_la: J_per_m3 {pub: in};
        var v_pas: m3_per_s {init: 0.0, pub: out};
        var v_pat: m3_per_s {init: 0.0, pub: out};
        var v_par: m3_per_s {init: 0.0, pub: out};
        var v_pcp: m3_per_s {init: 0.0, pub: out};
        var v_psh: m3_per_s {init: 0.0, pub: out};
        var v_pvn: m3_per_s {init: 0.0, pub: out};
        var v_puv: m3_per_s {pub: in};

        // Constitutive Relations

        ode(u_pas, t) = (v_puv-v_pas)/C_pas;
        ode(u_pat, t) = (v_pas-v_pat)/C_pat;
        ode(u_par, t) = (v_pat-v_psh-v_par)/C_par;
        ode(u_pcp, t) = (v_par-v_pcp)/C_pcp;
        ode(u_pvn, t) = (v_pcp+v_psh-v_pvn)/C_pvn;
        ode(v_pas, t) = (u_pas-u_pat-v_pas*R_pas)/I_pas;
        ode(v_pat, t) = (u_pat-u_par-v_pat*R_pat)/I_pat;
        ode(v_par, t) = (u_par-u_pcp-v_par*R_par)/I_par;
        ode(v_pcp, t) = (u_pcp-u_pvn-v_pcp*R_pcp)/I_pcp;
        ode(v_pvn, t) = (u_pvn-u_la-v_pvn*R_pvn)/I_pvn;
        ode(v_psh, t) = (u_par-u_pvn-v_psh*R_psh)/I_psh;
    enddef;

    def comp Heart as
        var t: second {pub: in};

        // Constitutive Parameters

        var mt: second;
        var e_a: dimensionless;
        var e_v: dimensionless;
        var T: second {priv: in};
        var t_ac: dimensionless {priv: in};
        var t_ar: dimensionless {priv: in};
        var T_ac: dimensionless {priv: in};
        var T_ar: dimensionless {priv: in};
        var T_vc: dimensionless {priv: in};
        var T_vr: dimensionless {priv: in};
        var R_trv: J_s_per_m6 {priv: in};
        var R_puv: J_s_per_m6 {priv: in};
        var R_miv: J_s_per_m6 {priv: in};
        var R_aov: J_s_per_m6 {priv: in};
        var B_trv: J_s2_per_m9 {priv: in};
        var B_puv: J_s2_per_m9 {priv: in};
        var B_miv: J_s2_per_m9 {priv: in};
        var B_aov: J_s2_per_m9 {priv: in};
        var L_trv: J_s2_per_m6 {priv: in};
        var L_puv: J_s2_per_m6 {priv: in};
        var L_miv: J_s2_per_m6 {priv: in};
        var L_aov: J_s2_per_m6 {priv: in};
        var E_lv_A: J_per_m6 {priv: in};
        var E_lv_B: J_per_m6 {priv: in};
        var E_la_A: J_per_m6 {priv: in};
        var E_la_B: J_per_m6 {priv: in};
        var E_rv_A: J_per_m6 {priv: in};
        var E_rv_B: J_per_m6 {priv: in};
        var E_ra_A: J_per_m6 {priv: in};
        var E_ra_B: J_per_m6 {priv: in};
        var q_ra_0: m3 {priv: in};
        var q_rv_0: m3 {priv: in};
        var q_la_0: m3 {priv: in};
        var q_lv_0: m3 {priv: in};

        // State Variables

        var u_ra: J_per_m3 {pub: out};
        var u_rv: J_per_m3;
        var u_la: J_per_m3 {pub: out};
        var u_lv: J_per_m3;
        var u_root: J_per_m3 {pub: in};
        var u_pas: J_per_m3 {pub: in};
        var v_trv: m3_per_s {init: 0};
        var v_puv: m3_per_s {init: 0, pub: out};
        var v_miv: m3_per_s {init: 0};
        var v_aov: m3_per_s {init: 0, pub: out};
        var v_sup_venacava: m3_per_s {pub: in};
        var v_inf_venacava: m3_per_s {pub: in};
        var v_pvn: m3_per_s {pub: in};
        var q_ra: m3 {init: 20.0e-6};
        var q_rv: m3 {init: 500.0e-6};
        var q_la: m3 {init: 20.0e-6};
        var q_lv: m3 {init: 500.0e-6};

        // Elastance Functions

        mt = t-T*floor(t/T);

        e_a = sel
            case mt <= (t_ar+T_ar)*T-T:
                0.5{dimensionless}*(1{dimensionless}+cos(pi*(mt+T-t_ar*T)/(T_ar*T)));
            case mt <= t_ac*T:
                0{dimensionless};
            case mt <= (t_ac+T_ac)*T:
                0.5{dimensionless}*(1{dimensionless}-cos(pi*(mt-t_ac*T)/(T_ac*T)));
            case mt <= T:
                0.5{dimensionless}*(1{dimensionless}+cos(pi*(mt-t_ar*T)/(T_ar*T)));
        endsel;

        e_v = sel
            case mt <= T_vc*T:
                0.5{dimensionless}*(1{dimensionless}-cos(pi*mt/(T_vc*T)));
            case mt <= (T_vc+T_vr)*T:
                0.5{dimensionless}*(1{dimensionless}+cos(pi*(mt-T_vc*T)/(T_vr*T)));
            case mt < T:
                0{dimensionless};
        endsel;

        // Constitutive Relations

        ode(v_trv, t) = sel
            case u_ra >= u_rv:
                (u_ra-u_rv-(R_trv+B_trv*abs(v_trv))*v_trv)/L_trv;
            case u_ra < u_rv:
                -(R_trv+B_trv*abs(v_trv))*v_trv/L_trv;
        endsel;

        ode(v_puv, t) = sel
            case u_rv >= u_pas:
                (u_rv-u_pas-(R_puv+B_puv*abs(v_puv))*v_puv)/L_puv;
            case u_rv < u_pas:
                -(R_puv+B_puv*abs(v_puv))*v_puv/L_puv;
        endsel;

        ode(v_miv, t) = sel
            case u_la >= u_lv:
                (u_la-u_lv-(R_miv+B_miv*abs(v_miv))*v_miv)/L_miv;
            case u_la < u_lv:
                -(R_miv+B_miv*abs(v_miv))*v_miv/L_miv;
        endsel;

        ode(v_aov, t) = sel
            case u_lv >= u_root:
                (u_lv-u_root-(R_aov+B_aov*abs(v_aov))*v_aov)/L_aov;
            case u_lv < u_root:
                -(R_aov+B_aov*abs(v_aov))*v_aov/L_aov;
        endsel;

        u_ra = (e_a*E_ra_A+E_ra_B)*(q_ra-q_ra_0);
        u_rv = (e_v*E_rv_A+E_rv_B)*(q_rv-q_rv_0);
        u_la = (e_a*E_la_A+E_la_B)*(q_la-q_la_0);
        u_lv = (e_v*E_lv_A+E_lv_B)*(q_lv-q_lv_0);

        // Conservation Laws

        ode(q_ra, t) = v_sup_venacava+v_inf_venacava-v_trv;
        ode(q_rv, t) = v_trv-v_puv;
        ode(q_la, t) = v_pvn-v_miv;
        ode(q_lv, t) = v_miv-v_aov;
    enddef;
